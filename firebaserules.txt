rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow Cloud Functions (service account) full access
    match /{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == null;
    }

    // Users collection: owners read self; admins read all.
    match /users/{userId} {
      allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
      allow create: if request.auth != null
        && request.auth.uid == userId
        && isSafeUserCreate();
      allow update: if request.auth != null && (
        isAdmin() ||
        (request.auth.uid == userId && (userSelfUpdateAllowed() || isProTrialSelfUpgrade()))
      );
      allow delete: if isAdmin();

      // Subcollection tiles: owners read/write when Pro; admins read/write all
      match /tiles/{tileId} {
        allow read: if request.auth != null && (
          isAdmin() ||
          request.auth.uid == userId
        );
        allow write: if request.auth != null && (
          isAdmin() ||
          (request.auth.uid == userId && isPro())
        );
      }

      // Subcollection saved_results: Allow Pro users to read/write their own results, admins to read/write all
      match /saved_results/{resultId} {
        allow read, write: if request.auth != null && (
          isAdmin() || 
          (request.auth.uid == userId && isPro())
        );
      }

      // Labour quotes: Pro users with labour add-on (client-gated); owner read/write
      match /labour_quotes/{quoteId} {
        allow read, write: if request.auth != null && (
          isAdmin() ||
          (request.auth.uid == userId && isPro())
        );
      }
    }

    // Main tile database
    match /tiles/{tileId} {
      allow read: if request.auth != null && isPro() && (
        resource.data.isPublic == true && 
        resource.data.isApproved == true
      );
      allow write: if request.auth != null && isAdmin();
    }

    // Collection group query for pending submissions (admins only)
    match /{path=**}/tiles/{tileId} {
      allow read: if request.auth != null && isAdmin();
    }

    // Collection group query for saved_results (admin analytics)
    match /{path=**}/saved_results/{resultId} {
      allow read: if request.auth != null && isAdmin();
    }

    // Company accounts
    match /organisations/{orgId} {
      allow read: if isOrgMember(orgId) || isAdmin();
      allow create: if request.auth != null
        && isPro()
        && request.resource.data.createdByUserId == request.auth.uid;
      allow update: if isOrgOwner(orgId) || isAdmin();

      match /members/{memberId} {
        allow read: if isOrgMember(orgId) || isAdmin();
        allow create: if request.auth != null && (
          isAdmin() ||
          (request.auth.uid == memberId && isPro()) ||
          isOrgOwner(orgId)
        );
        allow update, delete: if isOrgOwner(orgId) || isAdmin();
      }

      match /invites/{inviteId} {
        allow read: if isOrgOwner(orgId) || isAdmin();
        allow create: if isOrgOwner(orgId) || isAdmin();
        allow update: if isOrgOwner(orgId) || isAdmin();
      }

      match /assignments/{assignmentId} {
        allow read: if isOrgMember(orgId) || isAdmin();
        allow create: if isOrgEstimator(orgId) || isAdmin();
        allow update: if isOrgEstimator(orgId) || isAdmin()
          || (isOrgMember(orgId) && resource.data.assignedToUserId == request.auth.uid);
      }

      match /jobs/{jobId} {
        allow read: if isOrgMember(orgId) || isAdmin();
        allow create: if isOrgEstimator(orgId) || isAdmin();
        allow update: if isOrgEstimator(orgId) || isAdmin()
          || (isOrgMember(orgId) && resource.data.assignedToUserId == request.auth.uid);
        allow delete: if isOrgOwner(orgId) || isAdmin();
      }
    }

    // Pending invite lookup for joining users
    match /{path=**}/invites/{inviteId} {
      allow read, update: if request.auth != null
        && resource.data.email == request.auth.token.email.lower();
    }

    // Rules for sessions and calculations collections
    match /sessions/{sessionId} {
      allow read: if request.auth != null && isAdmin();
      allow write: if request.auth != null && request.auth.uid == sessionId;
    }

    match /calculations/{calculationId} {
      allow read: if request.auth != null && (request.auth.uid == resource.data.userId || isAdmin());
      allow create: if request.auth != null && isPro() && request.auth.uid == request.resource.data.userId;
      allow update: if request.auth != null && isPro() && 
                    request.auth.uid == resource.data.userId && 
                    request.resource.data.userId == resource.data.userId; // Prevent changing userId
      allow delete: if request.auth != null && isPro() && request.auth.uid == resource.data.userId;
    }

    // Helper functions
    function isDesignatedAdmin() {
      return request.auth != null && request.auth.token.email != null && (
        request.auth.token.email == 'support@roofgrid.uk' ||
        request.auth.token.email == 'hgwarner1307@gmail.com'
      );
    }

    function userDoc() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function userDocExists() {
      return request.auth != null &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function isAdmin() {
      return request.auth != null && (
        isDesignatedAdmin() ||
        (userDocExists() && userDoc().data.role == 'admin')
      );
    }

    function isPro() {
      return request.auth != null && (
        isDesignatedAdmin() ||
        (userDocExists() && (
          userDoc().data.role == 'pro' ||
          userDoc().data.role == 'admin'
        ))
      );
    }

    function isOrgMember(orgId) {
      return request.auth != null &&
        exists(/databases/$(database)/documents/organisations/$(orgId)/members/$(request.auth.uid));
    }

    function orgMemberRole(orgId) {
      return get(/databases/$(database)/documents/organisations/$(orgId)/members/$(request.auth.uid)).data.role;
    }

    function isOrgOwner(orgId) {
      return isOrgMember(orgId) && orgMemberRole(orgId) == 'owner';
    }

    function isOrgEstimator(orgId) {
      return isOrgMember(orgId) && (
        orgMemberRole(orgId) == 'owner' ||
        orgMemberRole(orgId) == 'estimator'
      );
    }

    function privilegedUserFields() {
      return [
        'role',
        'labourCalculatorActive',
        'customerQuoteActive',
        'stripeCustomerId',
        'subscriptionId',
        'subscriptionPlan',
        'subscriptionStatus',
        'subscriptionEndDate',
        'labourSubscriptionId',
        'labourSubscriptionPlan',
        'labourSubscriptionStatus',
        'labourSubscriptionEndDate',
        'customerQuoteSubscriptionId',
        'customerQuoteSubscriptionPlan',
        'customerQuoteSubscriptionStatus',
        'customerQuoteSubscriptionEndDate',
      ];
    }

    function isSafeUserCreate() {
      return request.resource.data.role == 'free'
        && request.resource.data.labourCalculatorActive != true
        && request.resource.data.customerQuoteActive != true;
    }

    function userSelfUpdateAllowed() {
      return !request.resource.data.diff(resource.data).affectedKeys()
          .hasAny(privilegedUserFields());
    }

    // One-time self-serve Pro trial start from the client upgrade flow.
    function isProTrialSelfUpgrade() {
      return resource.data.role == 'free'
        && request.resource.data.role == 'pro'
        && request.resource.data.proTrialStartDate is timestamp
        && request.resource.data.proTrialEndDate is timestamp
        && request.resource.data.diff(resource.data).affectedKeys()
            .hasOnly(['role', 'proTrialStartDate', 'proTrialEndDate', 'lastUpdated']);
    }
  }
}
